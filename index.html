Example Code 1: Quadratic Solver

import math

def quadratic_solver(a, b, c):
    d = b**2 - 4*a*c
    if d < 0:
        return "No real solutions"
    x1 = (-b + math.sqrt(d)) / (2*a)
    x2 = (-b - math.sqrt(d)) / (2*a)
    return x1, x2
